window.generateAndSharePDF = async () => { if(cachedLogs.length === 0) return alert("No data to export!"); const { jsPDF } = window.jspdf; const docPDF = new jsPDF(); const name = document.getElementById('u-name').value || "Not Specified"; const company = document.getElementById('u-comp').value || "Not Specified"; const total = document.getElementById('total-val').innerText; // تحويل قيمة Month Filter لاسم شهر مقروء const monthVal = document.getElementById('month-filter').value; // YYYY-MM const monthDate = new Date(monthVal + "-01"); const monthName = monthDate.toLocaleString('en-US', { month: 'long', year: 'numeric' }); // --- تصميم الهيدر (Header) --- docPDF.setFillColor(240, 240, 240); docPDF.rect(0, 0, 210, 40, 'F'); // خلفية رمادية خفيفة للهيدر docPDF.setFontSize(24); docPDF.setTextColor(0, 123, 255); docPDF.text("WORK REPORT", 105, 25, null, null, "center"); // --- معلومات الموظف والشركة --- docPDF.setFontSize(11); docPDF.setTextColor(50, 50, 50); docPDF.text("EMPLOYEE DETAILS:", 20, 50); docPDF.setFont(undefined, 'bold'); docPDF.text(`Name: ${name}`, 20, 57); docPDF.text(`Company: ${company}`, 20, 64); docPDF.setFont(undefined, 'normal'); docPDF.text("REPORT PERIOD:", 140, 50); docPDF.setFont(undefined, 'bold'); docPDF.text(`${monthName}`, 140, 57); // --- رسم خط فاصل --- docPDF.setDrawColor(200, 200, 200); docPDF.line(20, 70, 190, 70); // --- جدول البيانات (Table) --- docPDF.autoTable({ startY: 75, head: [['Date', 'Description', 'Hours Worked']], body: cachedLogs.map(item => [item[0], "Standard Work Session", item[1] + " Hrs"]), theme: 'striped', headStyles: { fillColor: [0, 123, 255], textColor: [255, 255, 255], fontSize: 12, halign: 'center' }, columnStyles: { 0: { halign: 'center', cellWidth: 40 }, 1: { halign: 'left' }, 2: { halign: 'center', cellWidth: 40 } }, styles: { fontSize: 10, cellPadding: 5 } }); // --- التذييل والخلاصة (Footer) --- const finalY = docPDF.lastAutoTable.finalY + 15; // مربع الإجمالي docPDF.setFillColor(0, 123, 255); docPDF.rect(130, finalY - 8, 60, 15, 'F'); docPDF.setTextColor(255, 255, 255); docPDF.setFontSize(13); docPDF.text(`TOTAL: ${total} HRS`, 160, finalY + 2, null, null, "center"); // توقيع بسيط docPDF.setTextColor(150, 150, 150); docPDF.setFontSize(9); docPDF.text("Generated by WorkTime Pro App", 105, 285, null, null, "center"); // --- المشاركة أو الحفظ --- const blob = docPDF.output('blob'); const file = new File([blob], `Report_${monthName}_${name}.pdf`, { type: "application/pdf" }); if (navigator.share) { try { await navigator.share({ files: [file], title: `Work Report - ${monthName}` }); } catch (e) { docPDF.save(`Report_${monthName}.pdf`); } } else { docPDF.save(`Report_${monthName}.pdf`); } };
